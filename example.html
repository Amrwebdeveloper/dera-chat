الاستخدام النهائي — سطرين

ملف واحد، وبلوك JSON واحد. الباقي يتولّاه البندل تلقائيًا.

<!-- 1) ملف واحد -->
<script src="chat-widget.min.js"></script>

<!-- 2) كل الإعدادات JSON -->
<script type="application/json" data-chat-widget>
{
  "identity": { "name": "عيادة الشفاء", "avatar": "logo.png" },
  "theme":    { "primary": "#0ea5e9", "radius": "20px" },
  "fab":      { "icon": "headset", "shape": "rounded" },
  "features": { "history": true, "suggestions": true },
  "responses": {
    "موقع|خريطة": { "type": "embed", "embed": { "src": "https://...optional-map..." } },
    "صورة|نصيحة": { "type": "image", "image": { "src": "tip.png", "caption": "نصيحة" } },
    "*":          { "type": "text",  "text": "أهلاً بك! كيف أساعدك؟" }
  }
}
</script>

البديل البرمجي (نفس الـJSON): ChatWidget.init({ ... }) — أو تحميل ملف خارجي chat-widget.json.
