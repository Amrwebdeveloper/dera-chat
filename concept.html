١) الإنشاء — تمرّر الخصائص

كائن واحد بكل الإعدادات، بما فيها معالج الردود onMessage.

const chat = new ChatWidget({
  el: '#container',                 // اختياري؛ افتراضيًا داخل <body>
  identity: { name: 'عيادة الشفاء', avatar: 'logo.png', status: 'متصل الآن' },
  theme:    { primary: '#0ea5e9', accent: '#06b6d4', radius: '22px' },
  layout:   { side: 'left', fabBottom: '1.5rem' },
  fab:      { icon: 'headset', shape: 'rounded' },
  send:     { icon: 'send', rotate: true },
  features: { history: true, suggestions: true },
  behavior: { welcome: 'أهلاً بك 👋', typeSpeed: 20 },

  // معالج الردود (يربط الباك-إند) — أو responses كخريطة JSON
  onMessage: async (text) => {
    const res = await fetch('/api/chat', { method:'POST', body: JSON.stringify({ text }) });
    return await res.json();        // { type:'text'|'image'|'carousel'|... }
  },
});

٢) الاستماع للأحداث (Listeners)

سلسلة قابلة للربط .on() / .off().

chat
  .on('ready',   ()      => console.log('جاهز'))
  .on('open',    ()      => console.log('فُتحت المحادثة'))
  .on('close',   ()      => console.log('أُغلقت'))
  .on('message', (text)  => console.log('المستخدم:', text))
  .on('reply',   (resp)  => console.log('رد البوت:', resp));

chat.off('open', handler);   // إلغاء الاستماع

events

readyالويدجت جاهز
open / closeفتح / إغلاق
messageأرسل المستخدم رسالة (text)
replyردّ البوت (response)

methods

open() · close() · toggle()تحكّم في الظهور
send(text)إرسال رسالة مستخدم
reply(response)دفع رد بأي قالب
setTheme · setIdentityتعديل لحظي
setFeatures · setLayoutتعديل لحظي
update(options)تحديث أي خصائص
clear()مسح المحادثة

properties

chat.isOpenهل المحادثة مفتوحة
chat.optionsالخصائص الحالية
